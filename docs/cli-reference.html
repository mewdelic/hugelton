CLI Referencecommand line tools

WebMCP で開く

arduino-cli — RP2040 / RP2350

arduino-cli compile -b rp2040:rp2040:rpipico \
  --board-options="usbstack=tinyusb" \
  --output-dir build/ \
  --export-binaries \
  /path/to/sketch

Wokwi CLI — Simulation (no browser)

export WOKWI_CLI_TOKEN="your-ci-token"
wokwi-cli /project --timeout 5000
[wokwi]
version = 1
firmware = 'build/sketch.ino.uf2'
elf = 'build/sketch.ino.elf'
rfc2217ServerPort = 4000

picotool — Flash / Reset

picotool load firmware.uf2           # flash UF2
picotool info                          # device info
picotool save -t elf firmware.elf      # dump flash → ELF

sigrok-cli — Logic Analyzer

sigrok-cli --driver=fx2lafw --channels=CH1,CH2,CH3 \
  --samples=1M -O vcd capture.vcd

himalaya — Email (IMAP / Gmail)

himalaya inbox list
himalaya inbox read --folder=INBOX --limit=10
